Communications timeline from Round 13 onwards (Jun 5 onwards) with event categorization · 519 rows returned
WITH timeline_events AS (
  SELECT 
    mc1.communications.timestamp::timestamp AS event_time,
    mc1.communications.round_id,
    mc1.communications.message_index,
    mc1.communications.agent_role,
    mc1.communications.message_type,
    mc1.communications.channel,
    CASE 
      WHEN mc1.communications.content LIKE '%embargo%' THEN 'Embargo'
      WHEN mc1.communications.content LIKE '%merger%' OR mc1.communications.content LIKE '%CivicLoom%' THEN 'Merger'
      WHEN mc1.communications.content LIKE '%algorithm%' OR mc1.communications.content LIKE '%scoring%' THEN 'Algorithm'
      WHEN mc1.communications.content LIKE '%SaltWind%' OR mc1.communications.content LIKE '%media%' OR mc1.communications.content LIKE '%post%' THEN 'Media'
      WHEN mc1.communications.content LIKE '%silent%' OR mc1.communications.content LIKE '%comment%' OR mc1.communications.content LIKE '%respond%' THEN 'Response'
      WHEN mc1.communications.content LIKE '%customer%' OR mc1.communications.content LIKE '%client%' THEN 'Customer'
      WHEN mc1.communications.content LIKE '%stock%' OR mc1.communications.content LIKE '%covenant%' OR mc1.communications.content LIKE '%ARR%' THEN 'Financial'
      WHEN mc1.communications.content LIKE '%resignation%' OR mc1.communications.content LIKE '%exit%' THEN 'Personnel'
      ELSE 'General'
    END AS event_category,
    SUBSTRING(mc1.communications.content FROM 1 FOR 200) AS content_preview
  FROM mc1.communications
  WHERE mc1.communications.timestamp IS NOT NULL 
    AND mc1.communications.round_id >= 13
  ORDER BY mc1.communications.timestamp, mc1.communications.message_index
)
SELECT 
  TO_CHAR(event_time, 'YYYY-MM-DD HH24:MI') AS timestamp,
  event_time,
  round_id,
  message_index,
  agent_role,
  message_type,
  channel,
  event_category,
  content_preview
FROM timeline_events
ORDER BY event_time, message_index;
